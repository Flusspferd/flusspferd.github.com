 24 /**
 25  * Create an iterable object returning all objects in a range, starting
 26  * at "from", up to (not including) "to", incrementing by "by" in each step.
 27  *
 28  * @param {number} from The start of the range.
 29  * @param {number} to The (non-inclusive) end of the range.
 30  * @param {number} by The increment.
 31  *
 32  * @class
 33  */
 34 function Range(from,to, by) {
 35   var i = from;
 36   by = by || 1;
 37   var r;
 38   function RangeInstance() {
 39     while (i < to) {
 40       r[i] = i;
 41       yield i;
 42       i += by;
 43     }
 44   };
 45   r = new RangeInstance();
 46   r.__iterator__ = function() { return r };
 47 
 48   return r;
 49 }
