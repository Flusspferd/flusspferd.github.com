 26 // Test for equality any JavaScript type.
 27 // Discussions and reference: http://philrathe.com/articles/equiv
 28 // Test suites: http://philrathe.com/tests/equiv
 29 // Author: Philippe RathÃ© <prathe@gmail.com>
 30 
 31 // About passing arguments:
 32 //      when < 2   : return true
 33 //      when 2     : return true if 1st equals 2nd
 34 //      when > 2   : return true 1st equals 2nd, and 2nd equals 3rd,
 35 //                      and 3rd equals 4th ... ans so on (by transition)
 36 //
 37 
 38 // Stack of constructors that enables us to skip or abort on functions value.
 39 var objectCallerConstructorStack = [];
 40 function equiv() {
 41     // we will need to use the array splice method,
 42     // so we need to convert arguments into a true array.
 43     var args = Array.prototype.slice.apply(arguments);
 44     var a, b; // compares a and b (1st and 2nd argument)
 45     var i; // for iterating over objects convenience
 46     var len; // for iterating array's length memoization
 47     // Memoized objects' properties.
 48     // Allow us to test for equivalence only in one way,
 49     // then to compare properties to make sure no one is missing.
 50     var aProperties, bProperties;
 51 
 52 
 53     if (args.length < 2) {
 54         return true; // nothing to compare together
 55     }
 56 
 57     a = args[0];
 58     b = args[1];
 59 
 60     return (function (a, b) {
 61 
 62         // Try to optimize the algo speed if ever both a and b are references
 63         // pointing to the same object (e..g. only for functions, arrays and objects)
 64         if (a === b) {
 65             return true;
 66         }
 67         
 68         // Must test for the null value before testing for the type of an object,
 69         // because null is also an object. Trapping null or undefined values here
 70         // is a good to avoid to much validations further in the code.
 71         if (a === null || b === null || typeof a === "undefined" || typeof b === "undefined") {
 72             return false; // Anyway (a === b) would have already caught it.
 73         }
 74 
 75         // Don't lose time and prevent further wrong type manipulation or unexpected results.
 76         if (typeof a !== typeof b) {
 77             return false;
 78         }
 79 
 80         // Must test if it's an array before testing for the type of an object,
 81         // because an array is also an object.
 82         if (a instanceof Array) {
 83 
 84             // Make sure b is also an array.
 85             // At this point b can be an object.
 86             // Prevent further arrays operation on object.
 87             if ( ! (b instanceof Array)) {
 88                 return false;
 89             }
 90 
 91             len = a.length;
 92             if (len !== b.length) { // safe and faster
 93                 return false;
 94             }
 95             for (i = 0; i < len; i++) {
 96                 if( ! equiv(a[i], b[i])) {
 97                     return false;
 98                 }
 99             }
100             return true;
101         }
102         
103         // NOTE:
104         //      Must test if it's a date before testing if it is an object,
105         //      because a date is also an object.
106         if (a instanceof Date) {
107             return b instanceof Date && a.valueOf() === b.valueOf();
108         }
109 
110         // NOTE:
111         //      Using toSource() to compare regexp only works in FF. It was working as well as the above.
112         //      Must verify the constructor of a regexp before verifying it is an object.
113         //      Notice those confusing true statements:
114         //          var a = /./;
115         //          a instanceof Object; // => true
116         //          a instanceof RegExp; // => true
117         //          typeof a === "function"; // => false in IE and Opera, true in FF and Safari
118         //
119         // There is 3 possible modifier for regular expressions (g, m and i)
120         // The source property only returns the regular expression string without the modifier.
121         if (a instanceof RegExp) {
122             return b instanceof RegExp &&
123                 a.source === b.source &&
124                 a.global === b.global &&
125                 a.ignoreCase === b.ignoreCase &&
126                 a.multiline === b.multiline;
127         }
128 
129         // typeof on some types returns sometimes unexpected type that we deals with.
130         // We could have used instanceof instead of typeof but it comes with
131         // other problems as mentioned above when testing for the RegExp constructor.
132         // We explicitely don't use hasOwnProperty when iterating on the properties of
133         // an object to allow a deeper equivalence (e.g. for instances particularly)
134         if (typeof a === "object") {
135             // Different constructors means a and b can't be equivalent instances.
136             // We don't have any choice of comparing them in both ways because of inheritance
137             // otherwise {} could be mistaken with []
138             // NOTE this:
139             //      var a = [];
140             //      a instanceof Array;  // true
141             //      a instanceof Object; // true
142             if ( a.constructor !== b.constructor) {
143                 return false;
144             }
145 
146             // Stack constructors before iterating over its properties
147             objectCallerConstructorStack.push(a.constructor);
148 
149             aProperties = [];
150             bProperties = [];
151 
152             // NOTE:
153             //      We only need to compare propertie's equivalence in one way
154             //      and ensures that all properties of a and b are the same.
155 
156             // Verify a's properties with b's properties equivalence.
157             // Stack a's properties
158             
159             for (i in a) {
160                 aProperties.push(i);
161                 if (!equiv(a[i], b[i])) {
162                     // Unstack current constructor when finished with the object
163                     objectCallerConstructorStack.pop();
164                     return false;
165                 }
166             }
167 
168             // Stack also b's properties
169             for (i in b) {
170                 bProperties.push(i);
171             }
172 
173             // Unstack current constructor when finished with the object
174             objectCallerConstructorStack.pop();
175 
176             // Ensures properties's name in both ways.
177             // Must sort them because they may not be always in the same order!
178             return equiv(aProperties.sort(), bProperties.sort());
179         }
180 
181         // NOTE:
182         //      Using the typeof operator will also catch a RegExp instance.
183         //      Take no risk.
184         if (a instanceof Function) {
185             // BEHAVIOR when comparing functions.
186             //      Being here means that a and b were were not the same references.
187             //      At this level, functions won't be compared at all.
188             //      - skip when they are properties of instances (not instantiates from the Object construtor)
189             //      - abort otherwise
190             //
191             //      To Determine if a function is anonymous we use a regexp on the function sources.
192             var currentObjectCallerConstructor =
193                     objectCallerConstructorStack[objectCallerConstructorStack.length - 1];
194             return currentObjectCallerConstructor !== Object &&
195                     typeof currentObjectCallerConstructor !== "undefined";
196         }
197 
198         // NaN is a number in JavaScript.
199         // Because this statement is false: 0/0 === 0/0,
200         // we must use the isNaN function which is the only way to know a number is NaN.
201         // Note that it isn't the case with the Infinity number, and that 1/0 === 2/0.
202         if (typeof a === "number" && isNaN(a)) {
203             return isNaN(b);
204         }
205 
206         // Compares Number, String or Boolean
207         // Because sooner we have already tried:
208         //          a === b (then return if it is the case)
209         //            and
210         //  typeof a !== typeof b (then return false if it is the case)
211         // at this time a and b should be of the same type, but hold different values.
212         // We can safely return false instead of returning a === b
213         return false;
214 
215     }(a, b)) && equiv.apply(this, args.splice(1, args.length -1)); // apply transition with (1..n) arguments
216 }

218 this.TestHarness = function TestHarness(name) {
219   this.name = name;
220   this.tests = [];
221   this.asserts_failed = 0;
222   this.total_asserts = 0;
223   this.tests_failed = 0;
224   this.total_tests = 0;
225   this.outputStream = IO.stdout;
226   TestHarness.suites.push(this);
227 }

229 TestHarness.suites = [];
230 
231 TestHarness.go = function TestHarness$go() {
232   var suites_failed = [];
233   var suites = Array.prototype.slice.apply(arguments);
234   if (suites.length == 0)
235     suites = this.suites;
236 
237   if (suites.length == 0)
238     return;
239 
240   for (var i in suites) {
241     var s = suites[i];
242     if (s.name === undefined)
243       s.name = 'Suite ' + i;
244     s.outputStream.print('#', s.name); 
245     
246     s.go();
247 
248     if (s.tests_failed > 0)
249       suites_failed.push(s);
250   }
251 
252   var msg;
253   if (suites_failed.length) {
254     msg = suites[0].red(suites_failed.length + '/' + suites.length, 'suites failed')
255   } else {
256     msg = suites[0].green(suites.length + '/' + suites.length, 'suites sucessful');
257   }
258   suites[0].outputStream.print(msg);
259 }

261 this.TestHarness.prototype = {
262   constructor: TestHarness,
263 
264   colourize: true,
265 
266   same: function() {
267     var args = Array.prototype.slice.call(arguments);
268 
269     var msg;
270     if (args.length > 2)
271       msg = args.pop();
272 
273     return this.do_assert( {
274       type: 'same',
275       when: new Date(),
276       ok: !!equiv.apply(equiv, args),
277       message: msg,
278       default_msg: "arguments are the same"
279     });
280   },
281 
282   instance_of: function(obj, type, msg) {
283     var test = obj instanceof type;
284     return this.do_assert( {
285       type: 'instance_of',
286       ok: !!test,
287       when: new Date(),
288       message: msg,
289       default_msg: "object is instance of type"
290     });
291   },
292 
293   ok: function ok(test, msg) {
294     if (arguments.length == 0) {
295       test = true; // If nothing is passed in, then assume its successful
296     }
297     return this.do_assert( {
298       type: 'ok',
299       ok: !!test,
300       when: new Date(),
301       message: msg,
302     } );
303   },
304 
305   // Handle/print the assert
306   do_assert: function do_assert(a) {
307     this.current_test.asserts.push(a);
308     a.num = this.current_test.asserts.length;
309 
310     // Padding to make numbers line up.
311     var padd = "  ";
312     var plan = this.current_test.plan;
313     
314     if (a.num === 1) {
315       // First assert this test
316       if (isNaN(plan)) {
317         // Just print the newline
318         this.outputStream.print();
319       }
320     }
321 
322     if (plan === undefined || isNaN(plan)) 
323       plan = a.num;
324 
325     let len = plan.toString().length - a.num.toString().length;
326     if (len > 0)
327       padd += Array(len+1).join(' ');
328 
329     let data = [padd + a.num];
330     data.push(a.ok ? this.green('ok') : this.red('not ok'));
331 
332     if (a.message !== undefined)
333       data.push('-', a.message)
334     else if ('default_msg' in a)
335       data.push('-', a.default_msg)
336 
337     if (a.ok)
338       this.current_test.asserts_passed++;
339     else 
340       this.current_test.asserts_failed++;
341 
342     this.outputStream.print(data);
343     return a.ok;
344   },
345 
346   expect: function expect(count) {
347     if (this.current_test.asserts.length)
348       throw new Error('expect called after an assertion');
349     this.current_test.plan = new Number(count).valueOf();
350     if (isNaN(this.current_test.plan))
351       throw new Error('invalid plan - not a number');
352     this.outputStream.print(" 1.." + this.current_test.plan);
353   },
354 
355   run: function run(name) {
356     this.current_test = { 
357       name: name, 
358       asserts: [],
359       asserts_passed: 0,
360       asserts_failed: 0,
361     };
362 
363     if (this[name] instanceof Function === false) {
364       throw new Error(name + " is not a function");
365     }
366 
367     this.tests.push(this.current_test);
368 
369     this.outputStream.write(name + ":");
370 
371     // If there's a 'plan' property of the function object, use it as the plan
372     if ('expect' in this[name]) {
373       this.expect(this[name].expect)
374     }
375 
376     try {
377         this[name].apply(this);
378     }
379     catch (e) {
380       this.current_test.error = e;
381     }
382     this.finalize_test();
383     delete this.current_test;
384   },
385 
386   go: function go() {
387     for (var n in this) {
388       if (this.hasOwnProperty(n) == false)
389         continue;
390       n = n.toString();
391 
392       if (n.substr(0,5) === "test_")
393         this.run(n);
394     }
395     this.finalize();
396   },
397 
398   green: function() {
399     var a = Array.prototype.slice.apply(arguments);
400     if (a.length == 0 && a[0] instanceof Array)
401       a = a[0];
402 
403     a = a.join(' ');
404     if (!this.colourize || !a.length)
405       return a;
406     return '\x1b[32m' + a + '\x1b[0m';
407   },
408 
409   red: function() {
410     var a = Array.prototype.slice.apply(arguments);
411     if (a.length == 1 && a[0] instanceof Array)
412       a = a[0];
413 
414     a = a.join(' ');
415     if (!this.colourize || !a.length)
416       return a;
417     return '\x1b[31m' + a + '\x1b[0m';
418   },
419 
420   finalize_test: function finalize_test() {
421     var test = this.current_test;
422     var msg = [];
423     this.asserts_failed += test.asserts_failed;
424 
425     if (test.plan !== null && !isNaN(test.plan)) {
426       // Check the plan.
427       let plural = test.plan == 1 ? 'assert' : 'asserts';
428 
429       if (test.plan > test.asserts.length) {
430         // too many
431         test.passed = false;
432         msg = [test.name, 'expected', test.plan, plural, 'only got', 
433                test.asserts.length]; 
434         this.total_asserts += test.plan;
435         this.asserts_failed += test.plan - test.asserts.length;
436       } else if (test.plan < test.asserts.length) {
437         test.passed = false;
438         msg = [test.name, 'expected', test.plan, plural, 'but got', 
439                test.asserts.length]; 
440         this.total_asserts += test.asserts.length;
441         this.asserts_failed += test.asserts.length - test.plan;
442       }
443       else {
444         test.passed = test.asserts_failed == 0;
445         this.total_asserts += test.plan;
446       }
447     } 
448 
449     if (test.error) {
450       // The test died.
451 
452       // Possibly before it could plan, so add a \n
453       if (isNaN(test.plan) && test.asserts.length == 0)
454         this.outputStream.print();
455       
456       let e = this.current_test.error;
457       this.outputStream.print( this.red(test.name, 'died:') );
458       var e_msg = e.toString().replace(/\n/g, "\n  ");
459       let a = ['  ' + e_msg];
460       if (e.fileName) {
461         var stack = e.stack;
462         a.push("\n  ", stack.replace(/\n/g, "\n  "));
463       }
464       msg = this.red(a);
465       test.passed = false;
466     } 
467     else {
468       // No plan, no error.
469       test.passed = test.asserts_failed == 0;
470     }
471     if (msg.length) {
472       this.outputStream.print(msg);
473     }
474 
475     this.total_tests++;
476     if (!test.passed)
477       this.tests_failed++;
478   },
479 
480   // Print summary
481   finalize: function finalize() {
482     if (this.tests_failed == 0) {
483       this.outputStream.print(this.green('All tests successful'))
484       return;
485     }
486 
487     var msg = [this.tests_failed + "/" + this.total_tests, "tests",
488               "(" + this.asserts_failed + "/" + this.total_asserts,
489               "asserts)", "failed"];
490 
491     this.outputStream.print(this.red(msg))
492   }
493 }
