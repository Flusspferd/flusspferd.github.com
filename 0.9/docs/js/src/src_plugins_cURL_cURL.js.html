 24 /**
 25  * Construct a cURL object. Takes no parameters
 26  * @name cURL
 27  * @constructor
 28  *
 29  * @class 
 30  * Network access via libcurl.
 31  *
 32  * == Example: ==
 33  * {{{
 34  * require('cURL')
 35  * c = new cURL();
 36  * c.url = 'http://www.google.com';
 37  * var status = c.perform();
 38  *
 39  * // Currently there is no auto-following of redirects, so:
 40  * while (status >= 300 && status <= 399) {
 41  *   c.url = c.header.Location;
 42  *   status = c.perform();
 43  * }
 44  * IO.stdout.write(c.responseBlob);
 45  * }}}
 46  *
 47  * Link to [[Blob]]
 48  */
 49 (function() {
 50   try {
 51     var cURL = exports.cURL; // get cURL as exported from the binary module
 52 
 53     // If we have HTTP/Headers, define a default header callback that
 54     // creates 
 55     require('HTTP/Headers');
 56 
 57     /** 
 58      * Array of protocols by the linked version of libcurl.
 59      * @name supportedProtocols
 60      * @fieldOf cURL
 61      */
 62     /** 
 63      * libcurl version number in hex. For example 7.18.2 is `0x071202`
 64      * @name versionHex
 65      * @fieldOf cURL
 66      * @type int
 67      */
 68     /** 
 69      * libcurl version number in dotted-decimal string form.
 70      * @name cURL#versionStr
 71      * @fieldOf cURL
 72      * @type string
 73      */
 74 
 75     /**
 76      * The URL to request
 77      * @name url
 78      * @fieldOf cURL.prototype
 79      */
 80     /**
 81      * HTTP Protocol version. Unknown/undefined for non-HTTP requests. See 
 82      * [[#headerReceived]] for caveats.
 83      * @name protocol
 84      * @fieldof cURL.prototype
 85      * @type strings
 86      */
 87     /** 
 88      * Numeric status code. Might not be defined for non-HTTP requests. See
 89      * [[#headerReceived]] for caveats.
 90      * @name status
 91      * @fieldof cURL.prototype
 92      * @type int
 93      */
 94     /** 
 95      * Human readable status message. Might not be defined for non-HTTP
 96      * requests. See [[#headerReceived]] for caveats.
 97      * @name statusMessage
 98      * @fieldof cURL.prototype
 99      * @type string
100      */
101     /**
102      * Binary response content. If you replace the
103      * [[#dataReceived]] callback than you will have to update this blob
104      * yourself.
105      * @name responseBlob
106      * @fieldOf cURL.prototype
107      * @type Blob
108      */
109 
110     /**
111      * HTTP response headers (if applicable). See [[#headerReceived]] for
112      * caveats.
113      * @name headers
114      * @fieldOf cURL.prototype
115      * @type HTTP.Headers
116      */
117 
118 
119     /**
120      * Set the request method to use when [[#perform]] is called. Currently understood
121      * values are "GET", "POST", "PUT" and "HEAD".
122      * @name setMethod
123      * @methodOf cURL.prototype
124      *
125      * @param {string} method request method
126      */
127 
128     /**
129      * Called for each line of the headers. The line includes the new line
130      * characters, and the end of headers are signified by "\r\n".<br /><br />
131      *
132      * Populates the [[#headers]], [[#protocol]], [[#status]] and
133      * [[#statusMessage]] properties after all headers have been received.<br
134      * /><br />
135      *
136      * This default implementation is conditionally provided on being able to
137      * load the [[HTTP.Headers]] module. If it cannot be loaded then the
138      * [[#headers]], [[#protocol]], [[#status]] and [[#statusMessage]]
139      * properties will not be defined.
140      *
141      * @param line a single line of header data
142      * @event
143      */
144     cURL.prototype.headerReceived = function headerReceived(hdr) {
145       if (hdr.match(/^\r\n/)) {
146         // Empty line signals end of headers
147         try {
148           // Pull out HTTP status line.
149           var match = /^(HTTP\/1.[10])\s+(\d+)\s+(.*?)\r\n/.exec(this.header_buffer);
150           if (match) {
151             this.header_buffer = this.header_buffer.substr(match[0].length);
152             [undefined, this.protocol, this.status, this.statusMessage] = match;
153           }
154 
155           this.headers = HTTP.Headers.parse(this.header_buffer);
156           delete this.header_buffer;
157         }
158         catch (e) 
159         {
160         } // Do nothing on error
161       }
162       else
163         this.header_buffer += hdr;
164     }
165   }
166   catch (e if /Unable to find library/.exec(e))
167   { // Stomp on the module not found message
168   }
169 
170   /**
171    * Called when a chunk of body data is available. The default implementation
172    * simply appends each chunk of data into [[#responseBlob]].
173    * @event
174    *
175    * @param {Blob} blob chunk of response data
176    */
177   cURL.prototype.dataReceived = function (blob) {
178     this.responseBlob.append(blob);
179   }
180 
181   if (cURL.prototype.perform.old === undefined) {
182     var old_perform = cURL.prototype.perform;
183     /**
184      * Perform the request on [[#url]].
185      *
186      * Before the request starts the [[#headers]], [[#protocol]], [[#status]],
187      * [[#statusMessage]] and the [[#responseBlob]] properties are cleared
188      */
189     cURL.prototype.perform = function perform() {
190       this.header_buffer = "";
191       delete this.headers;
192       delete this.protocol;
193       delete this.status;
194       delete this.statusMessage;
195       this.responseBlob = new Blob(0);
196       return perform.old.apply(this, arguments);
197     }
198     cURL.prototype.perform.old = old_perform;
199   }
200 
201   return cURL;
202 })()
