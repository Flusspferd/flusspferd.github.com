  6 // About passing arguments:
  7 //      when < 2   : return true
  8 //      when 2     : return true if 1st equals 2nd
  9 //      when > 2   : return true 1st equals 2nd, and 2nd equals 3rd,
 10 //                      and 3rd equals 4th ... ans so on (by transition)
 11 
 12 // Hoozit
 13 // Determine what is o.
 14 function hoozit(o) {
 15     if (typeof o === "string") {
 16         return "string";
 17 
 18     } else if (typeof o === "boolean") {
 19         return "boolean";
 20 
 21     } else if (typeof o === "number") {
 22 
 23         if (isNaN(o)) {
 24             return "nan";
 25         } else {
 26             return "number";
 27         }
 28 
 29     } else if (typeof o === "undefined") {
 30         return "undefined";
 31 
 32     // consider: typeof null === object
 33     } else if (o === null) {
 34         return "null";
 35 
 36     // consider: typeof [] === object
 37     } else if (o instanceof Array) {
 38         return "array";
 39     
 40     // consider: typeof new Date() === object
 41     } else if (o instanceof Date) {
 42         return "date";
 43 
 44     // consider: /./ instanceof Object;
 45     //           /./ instanceof RegExp;
 46     //          typeof /./ === "function"; // => false in IE and Opera,
 47     //                                          true in FF and Safari
 48     } else if (o instanceof RegExp) {
 49         return "regexp";
 50 
 51     } else if (typeof o === "object") {
 52         return "object";
 53 
 54     } else if (o instanceof Function) {
 55         return "function";
 56     }
 57 }
 58 
 59 // Call the o related callback with the given arguments.
 60 function bindCallbacks(o, callbacks, args) {
 61     var prop = hoozit(o);
 62     if (prop) {
 63         if (hoozit(callbacks[prop]) === "function") {
 64             return callbacks[prop].apply(callbacks, args);
 65         } else {
 66             return callbacks[prop]; // or undefined
 67         }
 68     }
 69 }
 70 
 71 var equiv = function () {
 72 
 73     var innerEquiv; // the real equiv function
 74     var callers = []; // stack to decide between skip/abort functions
 75 
 76     
 77     var callbacks = function () {
 78 
 79         // for string, boolean, number and null
 80         function useStrictEquality(b, a) {
 81             return a === b;
 82         }
 83 
 84         return {
 85             "string": useStrictEquality,
 86             "boolean": useStrictEquality,
 87             "number": useStrictEquality,
 88             "null": useStrictEquality,
 89             "undefined": useStrictEquality,
 90 
 91             "nan": function (b) {
 92                 return isNaN(b);
 93             },
 94 
 95             "date": function (b, a) {
 96                 return hoozit(b) === "date" && a.valueOf() === b.valueOf();
 97             },
 98 
 99             "regexp": function (b, a) {
100                 return hoozit(b) === "regexp" &&
101                     a.source === b.source && // the regex itself
102                     a.global === b.global && // and its modifers (gmi) ...
103                     a.ignoreCase === b.ignoreCase &&
104                     a.multiline === b.multiline;
105             },
106 
107             // - skip when the property is a method of an instance (OOP)
108             // - abort otherwise,
109             //   initial === would have catch identical references anyway
110             "function": function () {
111                 var caller = callers[callers.length - 1];
112                 return caller !== Object &&
113                         typeof caller !== "undefined";
114             },
115 
116             "array": function (b, a) {
117                 var i;
118                 var len;
119 
120                 // b could be an object literal here
121                 if ( ! (hoozit(b) === "array")) {
122                     return false;
123                 }
124 
125                 len = a.length;
126                 if (len !== b.length) { // safe and faster
127                     return false;
128                 }
129                 for (i = 0; i < len; i++) {
130                     if( ! innerEquiv(a[i], b[i])) {
131                         return false;
132                     }
133                 }
134                 return true;
135             },
136 
137             "object": function (b, a) {
138                 var i;
139                 var eq = true; // unless we can proove it
140                 var aProperties = [], bProperties = []; // collection of strings
141 
142                 // comparing constructors is more strict than using instanceof
143                 if ( a.constructor !== b.constructor) {
144                     return false;
145                 }
146 
147                 // stack constructor before traversing properties
148                 callers.push(a.constructor);
149 
150                 for (i in a) { // be strict: don't ensures hasOwnProperty and go deep
151 
152                     aProperties.push(i); // collect a's properties
153 
154                     if ( ! innerEquiv(a[i], b[i])) {
155                         eq = false;
156                     }
157                 }
158 
159                 callers.pop(); // unstack, we are done
160 
161                 for (i in b) {
162                     bProperties.push(i); // collect b's properties
163                 }
164 
165                 // Ensures identical properties name
166                 return eq && innerEquiv(aProperties.sort(), bProperties.sort());
167             }
168         };
169     }();
170 
171     innerEquiv = function () { // can take multiple arguments
172         var args = Array.prototype.slice.apply(arguments);
173         if (args.length < 2) {
174             return true; // end transition
175         }
176 
177         return (function (a, b) {
178             if (a === b) {
179                 return true; // catch the most you can
180 
181             } else if (typeof a !== typeof b || a === null || b === null || typeof a === "undefined" || typeof b === "undefined") {
182                 return false; // don't lose time with error prone cases
183 
184             } else {
185                 return bindCallbacks(a, callbacks, [b, a]);
186             }
187 
188         // apply transition with (1..n) arguments
189         })(args[0], args[1]) && arguments.callee.apply(this, args.splice(1, args.length -1));
190     };
191 
192     return innerEquiv;
193 
194 }();
195 
196 exports.equiv = equiv;
