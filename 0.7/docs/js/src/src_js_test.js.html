 29 function merge(ctor, obj) {
 30   for (let [k,v] in Iterator(obj))
 31     ctor[k] = v;

 34 const TAPProducer = function TAPProducer() {
 35   // TODO: ServerJS compliance
 36   this.outputStream = require('system').stdout;
 37 
 38   var depth;
 39   Object.defineProperty(this, "paddDepth", {
 40 
 41     getter: function() { return depth },
 42     setter: function(d) {
 43       d = new Number(d).valueOf();
 44       if (isNaN(d))
 45         throw TypeError("paddDepth must be a number");
 46       depth = d;
 47       this.padding = this.getPadd();
 48     },
 49   });
 50 
 51   this.paddDepth = 0;
 52 }

 55 
 56 merge(TAPProducer.prototype, {
 57   paddStr: '  ',
 58 
 59   colourize: true,
 60 
 61   green: function green() {
 62     var a = Array.slice.apply(arguments);
 63     if (a.length == 0 && a[0] instanceof Array)
 64       a = a[0];
 65 
 66     a = a.join(' ');
 67     if (!this.colourize || !a.length)
 68       return a;
 69     return '\x1b[32m' + a + '\x1b[0m';
 70   },
 71 
 72   red: function red() {
 73     var a = Array.slice.apply(arguments);
 74     if (a.length == 1 && a[0] instanceof Array)
 75       a = a[0];
 76 
 77     a = a.join(' ');
 78     if (!this.colourize || !a.length)
 79       return a;
 80     return '\x1b[31m' + a + '\x1b[0m';
 81   },
 82 
 83   // Handle/print the assert
 84   do_assert: function do_assert(a) {
 85     
 86     a.num = ++this._state.currentCase.numAsserts;
 87 
 88     let data = [a.num];
 89     data.push(a.ok ? this.green('ok') : this.red('not ok'));
 90 
 91     if (a.message !== undefined)
 92       data.push('-', a.message)
 93     else if ('defaultMsg' in a)
 94       data.push('-', a.defaultMsg)
 95 
 96     if (!a.ok)
 97       this._state.currentCase.assertsFailed.push(a);
 98 
 99     this.print(data);
100 
101     if (a.diag)
102       this.diag(a.diag);
103     if (a.errorDiag)
104       this.diag(a.errorDiag, this.red);
105     return a.ok;
106   },
107 
108   print: function print() {
109     var msg;
110     if (arguments[0] instanceof Array)
111       msg = arguments[0].join(" ");
112     else
113       msg = Array.join(arguments, " ");
114 
115     this.outputStream.print(this.padding + msg)
116   },
117 
118   diag: function diag(msg,filter) {
119     var self = this,
120         lines = msg.split(/\n/)
121 
122     lines = lines.map(function(l) {
123       return self.padding + ("# " + l)
124     });
125 
126     if (filter)
127       lines = lines.map(function(x) { return filter.call(self,x) });
128     lines = lines.join("\n");
129 
130     this.outputStream.print(lines);
131   },
132 
133   getPadd: function getPadd() {
134     var i = 0;
135     return [this.paddStr for (i in Util.Range(0, this.paddDepth))].join('');
136   }
137 });

140 const Suite = function Suite(cases) {
141   this.__proto__.__proto__.constructor.call(this);
142   this._state = {
143     cases: [],
144     casesFailed: 0,
145     assertsFailed: [],
146     numAsserts: 0,
147     run: 0,
148     currentCase: undefined
149   }
150 
151   if (!cases) {
152     return this;
153   }
154 
155 
156   // TODO: setup and teardown methods
157   for ([k,t] in Iterator(cases)) {
158 
159     if (/^test_/.test(k) == false)
160       continue;
161     this._state.cases.push({
162       func: t,
163       name: k.replace(/^test_(.*)/, "$1"),
164       index: this._state.cases.length,
165       numAsserts: 0,
166       assertsFailed: []
167     });
168   }
169 
170   this._state.plan = this._state.cases.length;
171 }
172 
173 Suite.prototype.__proto__ = TAPProducer.prototype;
174 
175 merge(Suite.prototype, {
176 
177   run: function run() {
178     if (!this._state.cases.length) {
179       throw new Error("Cannot run a Test.Suite without any test cases!");
180     }
181 
182     var oldCurrent = exports.__currentSuite__;
183     try {
184       exports.__currentSuite__ = this;
185       for ([name,test] in Iterator(this._state.cases)) {
186         this.runCase( test);
187       }
188     }
189     finally {
190       currentSuite = oldCurrent;
191     }
192     this.finalize();
193   },
194 
195   addSubSuite: function addSubSuite(s) {
196   },
197 
198   runCase: function runCase(theCase) {
199     var asserts = exports.asserts,
200         depth = this.paddDepth,
201         error;
202 
203     this.printPlan(this._state);
204 
205     try {
206       this.diag(theCase.name + "...");
207       this.paddDepth = depth + 1;
208       try {
209         this._state.currentCase = theCase;
210         theCase.func.call(asserts, this);
211         this.printPlan(theCase);
212       }
213       finally {
214         this.paddDepth = depth;
215       }
216     }
217     catch (e) {
218       error = e;
219     }
220     this.finishCase(theCase, error)
221   },
222 
223   finishCase: function finishCase(theCase, error) {
224     var failed = error || theCase.assertsFailed.length > 0;
225     this._state.currentCase = this._state;
226     if (error) {
227 
228       var diag = "case terminated with exception:\n"
229                + error.toString();
230 
231       if (error.fileName) {
232         diag += "\n  " +  error.stack.replace(/\n/g, "\n  ");
233       }
234 
235       this.do_assert( {
236         type: 'case',
237         when: new Date(),
238         errorDiag: diag,
239         ok: !failed,
240         message: theCase.name,
241       });
242       return;
243     }
244 
245     this.do_assert( {
246       type: 'case',
247       when: new Date(),
248       ok: !failed,
249       message: theCase.name,
250     });
251     delete this._state.currentCase;
252 
253   },
254 
255   printPlan: function printPlan(theCase) {
256     if (theCase.planned)
257       return;
258 
259     if ("plan" in theCase == false)
260       theCase.plan = theCase.numAsserts;
261 
262     this.print("1.." + theCase.plan);
263     theCase.planned = true;
264   },
265 
266   finalize: function finalize() {
267     //this.diag(this._state.toSource());
268   }
269 })

271 exports.Suite = Suite;
272 exports.runner = function runner() {
273   var suite;
274   var args = Array.slice(arguments);
275   if (args.length == 1 && args[0] instanceof Suite) {
276     builder = args[0];
277   }
278   else if (args.every(function(x) { return x instanceof Suite })) {
279     suite = new Suite();
280     args.forEach(suite.addSubSuite, builder);
281   }
282   else if (args[0] instanceof Object) {
283     suite = new Suite(args[0]);
284   }
285   else {
286     throw new TypeError("Invalid arguments to test.runner");
287   }
288 
289   return suite.run();
290 };

297       return currentSuite = x;
298     throw new TypeError("__currentSuite__ must be an instanceof test.Suite");
299   },
300   configurable: false,
301   enumerable: false,

307 const Asserts = function() {
308 }
309 
310 merge(Asserts.prototype, {
311   same: function() {
312     var args = Array.slice(arguments),
313         msg;
314 
315     if (args.length > 2)
316       msg = args.pop();
317     var ok = !!equiv.apply(equiv, args);
318 
319     var a = {
320       type: 'same',
321       when: new Date(),
322       ok: ok,
323       message: msg,
324       defaultMsg: "arguments are the same",
325     };
326 
327     if (!ok) {
328       a.wanted = args[0];
329       a.got = args.slice(1);
330       a.diag = "Wanted: " + (args[0] === undefined
331                              ? "undefined"
332                              : args[0].toSource())
333 
334                           + args.slice(1).map(function(i) {
335                               return "\n   Got: "
336                                    + (i === undefined
337                                           ? "undefined"
338                                           : i.toSource()
339                                      )
340                             });
341     }
342     return exports.__currentSuite__.do_assert(a);
343   },
344 
345   instanceOf: function(obj, type, msg) {
346     var test = obj instanceof type;
347     return exports.__currentSuite__.do_assert( {
348       type: 'instanceof',
349       ok: !!test,
350       when: new Date(),
351       message: msg,
352       defaultMsg: "is instanceof "
353                   + (type.name || "correct type")
354     });
355   },
356 
357   ok: function ok(test, msg) {
358     if (arguments.length == 0) {
359       test = true; // If nothing is passed in, then assume its successful
360     }
361     return exports.__currentSuite__.do_assert( {
362       type: 'ok',
363       ok: !!test,
364       when: new Date(),
365       message: msg,
366     } );
367   },
368 
369   expects: function expects(count) {
370     return exports.__currentSuite__.expects(count);
371   },
372 
373   diag: function diag() {
374     var suite = exports.__currentSuite__;
375     suite.diag.apply(suite, arguments);
376   }
377 
378 });
